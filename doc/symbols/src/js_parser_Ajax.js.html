1 /**
  2  * Handle the Regular Expression for Server roundtrips
  3  *
  4  * @author Georg Limbach <georf@dev.mgvmedia.com>
  5  *
  6  * @class ParserAjax
  7  * @constructor
  8  * @param {Function} config The config RegularExpression object
  9  * @param {String} text Text to parse
 10  * @param {String} expression Regular expression
 11  * @param {String} flags Flags to use
 12  * @param {Function} callback Callback function for data result
 13  */
 14 function ParserAjax(config, text, expression, flags, callback) {
 15 	this._URL = 'parser/parser.php';
 16 
 17 
 18 	this._config = config;
 19 	this._text = text;
 20 	this._expression = expression;
 21 	this._flags = flags;
 22 	this._callback = callback;
 23 
 24 
 25 	// post to server parser
 26 	$.post(this._URL, {
 27 			json: $.toJSON({
 28 				"parser": this._config.getName(),
 29 				"regularExpression": this._expression,
 30 				"flags": this._flags,
 31 				"matchText": this._text
 32 			})
 33 		}, callback, "json");
 34 }
