  3 var Permalink = function($link) {
  4     "use strict";
  5 
  6     // pointer to myself
  7     var self = this;
  8 
  9     this.update = function() {
 10 
 11         // counter, index and lenth
 12         var i, l;
 13 
 14         // current regex parser
 15         var parser = RegHex.getRegularExpression();
 16 
 17         // current regex flags
 18         var flags = parser.getSetFlags();
 19 
 20         // current regex
 21         var regex = parser.getRegularExpression();
 22 
 23         // current match texts as array
 24         var matchtexts = RegHex.getMatchTexts();
 25 
 26 
 27 
 28         // build a query
 29         var query = '?r=' + encodeURIComponent(regex) + '&f=';
 30 
 31         // add flags
 32         for (i = 0, l = flags.length; i < l; i++) {
 33             query += encodeURI(flags[i]);
 34             if (i+1 < l) {
 35                 query += encodeURIComponent(',');
 36             }
 37         }
 38 
 39         // add every matchtext
 40         for (i = 0, l = matchtexts.length; i < l; i++) {
 41             query += '&m' + i + '=' + encodeURIComponent(matchtexts[i].getText());
 42         }
 43 
 44         // add hash
 45         query += '#p-' + parser.getName();
 46 
 47         // put into link
 48         $link.attr('href', query);
 49     };
 50 
 51 
 52     // register to every change
 53     RegHex.registerChangeHandler(function() {
 54         self.update();
 55     });
 56 };
