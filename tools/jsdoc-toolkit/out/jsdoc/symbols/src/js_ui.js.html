  9 // unique counter for block id
 10 var matchingBlockId = 1;
 11 
 12 // pointer to last inserted match text
 13 var lastInsertedMatchText = null;
 14 
 15 $(function() {
 16     "use strict";
 17 
 18     // counter, length
 19     var i, l;
 20 
 21     // reset textarea and checkboxes(for firefox)
 22     $('textarea').val('');
 23 
 24     // Register existing match text fields
 25     $('.matchtext-block').each(function() {
 26         lastInsertedMatchText = new UIMatchText($(this), false);
 27     });
 28 
 29     // Create and register a message service
 30     RegHex.registerMessageService(new UIMessageService());
 31 
 32     // Create and register a code block service
 33     var codeSnip = new UICodeSnip();
 34     RegHex.registerCodeSnip(codeSnip);
 35 
 36     // Bundle flags into an array
 37     var bundleFlags = function() {
 38         var flags = [];
 39         $('.parser-option').each(function(i, el) {
 40             if ($(el).is(':checked')) {
 41                 flags.push($(el).attr('name').substr(7));
 42             }
 43         });
 44         return flags;
 45     };
 46 
 47     // Notify if regular expression changes
 48     $('#regex').bind('keyup blur input cut paste', function() {
 49         RegHex.updateRegularExpression($(this).val(), bundleFlags());
 50     });
 51 
 52     $('#add-matchtext').click(function() {
 53         // generate new id
 54             var newid = matchingBlockId++;
 55 
 56             // clone block
 57             var block = $('#matching-blocks').find('.matchtext-block').first()
 58                     .clone();
 59             block.find('label').html(' ').attr('for', 'newid' + newid);
 60             block.find('.textarea').text('').attr('id', 'newid' + newid);
 61             block.find('.matchtext-div').html(' ');
 62 
 63             // handle events with object
 64             lastInsertedMatchText = new UIMatchText(block, true);
 65         });
 66 
 67     var parserType = $('#parser-type');
 68 
 69     // Add parsers to select field
 70     for (i = 0, l = config.parsers.length; i < l; i++) {
 71         parserType.append($('<option value="' + config.parsers[i].getName() + '">'
 72                 + config.parsers[i].getName() + '</option>'));
 73     }
 74 
 75     // bind parser type to RegHex
 76     parserType.change(function() {
 77         var i, le;
 78 
 79 
 80         // update regex
 81         RegHex.changeParserType($('#parser-type').val());
 82 
 83         // update about window
 84         $('.current-parser-name').text($('#parser-type').val());
 85 
 86         // delete old flags
 87         $('#regex-options').html('');
 88 
 89         // enable usefull options
 90         var o = RegHex.getRegularExpression().getFlags();
 91         for (i = 0, le = o.length; i < le; i++) {
 92             $('#regex-options').append(o[i].getLi());
 93         }
 94 
 95         // Trigger regex change whenever an option changes
 96         $('#regex-options input[type="checkbox"]').change(function() {
 97             $('#regex').trigger('keyup');
 98         });
 99 
100         // set new info link
101         var l = RegHex.getRegularExpression().getUrls();
102         var ul = '';
103         for (i = 0, le = l.length; i < le; i++) {
104             ul += '<li><a href="' + l[i][0] + '">' + l[i][1] + '</a></li>';
105         }
106         if (ul != '') {
107             $('#additional-parser-info').html('<ul>' + ul + '</ul>');
108         } else {
109             $('#additional-parser-info').html(ul);
110         }
111 
112         codeSnip.update('');
113 
114         // trigger a new parse event
115         $('#regex').keyup();
116 
117         // update uri hash
118         window.location.hash = '#p-' + $('#parser-type').val();
119 
120     });
121 
122     // bind Help class to link
123     $('#button-tour').click(function() {
124         new Help();
125     return false;
126     });
127 
128 
129     // bind permalink handler
130     new Permalink($('#button-permalink'));
131 
132 
133     /*
134      * catch parser type from uri hash
135      * url should end as follow:
136      *  #p-javascript
137      * or
138      *  #p-sun-java
139      */
140     var match;
141     if (match = window.location.hash.match(/^#p-(.+)$/)) {
142         parserType.find('option').each(function() {
143             var self = $(this);
144             if (self.val() == match[1]) {
145                 self.attr('selected', true);
146             } else {
147                 self.removeAttr('selected');
148             }
149         });
150     }
151 
152     // trigger first change
153     parserType.change();
154 
155 
156     /*
157      * handle permalinks
158      *
159      * a permalink have some parts:
160      *      r
161      *      => regex {String}
162      *
163      *      f
164      *      => flags {String[]}
165      *
166      *      m0, m1, m2
167      *      => matchtexts {String}
168      *
169      * first parse search parameter for values
170      */
171     var parm = window.location.search.substring(1).split('&'),
172         match, first = true;
173 
174     for (i = 0, l = parm.length; i < l; i++) {
175         var parts = parm[i].split('=');
176         if (parts.length == 2 && parts[1] != '') {
177 
178             // flag part
179             if (parts[0] == 'f') {
180                 var flags = parts[1].split(',');
181                 for (var z = 0; z < flags.length; z++) {
182                     $('#option-' + decodeURIComponent(flags[z])).attr('checked', true);
183                 }
184 
185             // regex part
186             } else if (parts[0] == 'r') {
187                 $('#regex').val(decodeURIComponent(parts[1]));
188 
189             // matchtext part
190             } else if (match = parts[0].match(/^m(\d)$/)) {
191                 if (first) {
192                     first = false;
193                 } else {
194                     $('#add-matchtext').click();
195                 }
196                 lastInsertedMatchText.setText(decodeURIComponent(parts[1]));
197             }
198         }
199     }
200 
201     // update first time
202     $('#regex').trigger('keyup');
203 });
