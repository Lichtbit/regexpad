1 /**
  2  * A helper for real time typing
  3  *
  4  * @author Georg Limbach <georf@dev.mgvmedia.com>
  5  *
  6  *
  7  * @constructor
  8  * @param {jQuery} $area Textarea to write into
  9  * @param {String} _val Value to type
 10  * @param {int} _millisec Milliseconds between keys
 11  */
 12 function TypeInto($area, _val, _millisec) {
 13     "use strict";
 14 
 15 
 16     /**
 17      * Counter for steps
 18      * @type int
 19      */
 20     var _step = 0;
 21 
 22 
 23     /**
 24      * Type one key
 25      */
 26     var _typeKey = function() {
 27         $area.val(_val.substring(0, _step));
 28         $area.trigger('keyup');
 29 
 30         if (_step < _val.length) {
 31             _step++;
 32             setTimeout(_typeKey, _millisec);
 33         }
 34     };
 35 
 36     // begin to type
 37     _typeKey();
 38 }
