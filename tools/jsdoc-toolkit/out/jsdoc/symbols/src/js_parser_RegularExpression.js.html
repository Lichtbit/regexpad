1 /**
  2  * Handle Regular Expression for config
  3  *
  4  * @author Georg Limbach <georf@dev.mgvmedia.com>
  5  *
  6  * @class RegularExpression
  7  * @constructor
  8  * @param {String} _name Parser name
  9  * @param {Flag[]} _flags Allowed flags
 10  * @param {String[][]} _urls Array of info urls
 11  * @param {Function} _parser function to parse
 12  */
 13 function RegularExpression(_name, _flags, _urls, _parser) {
 14     "use strict";
 15 
 16 
 17     /**
 18      * Regular expression to parse
 19      * @type String
 20      */
 21     var _expression = '';
 22 
 23 
 24     /**
 25      * Set flags
 26      * @param String
 27      */
 28     var _setFlags = [];
 29 
 30 
 31     /**
 32      * Returns the name
 33      * @returns {String} name
 34      */
 35     this.getName = function() {
 36         return _name;
 37     };
 38 
 39 
 40     /**
 41      * Returns the allowed flags
 42      * @returns {Flags[]} flags
 43      */
 44     this.getFlags = function() {
 45         return _flags;
 46     };
 47 
 48 
 49     /**
 50      * Returns information urls
 51      * @returns {String[][]} urls
 52      */
 53     this.getUrls = function() {
 54         return _urls;
 55     }
 56 
 57 
 58     /**
 59      * Sets a new regular expression
 60      * @param {String} expression e.g. "[a-z]+"
 61      * @returns this
 62      */
 63     this.setRegularExpression = function (expression) {
 64         _expression = expression;
 65         return this;
 66     };
 67 
 68 
 69     /**
 70      * Sets the expression flags
 71      * @param {String[]} flags e.g. ["i","g"]
 72      * @returns this
 73      */
 74     this.setFlags = function (flags) {
 75         _setFlags = flags;
 76         return this;
 77     };
 78 
 79 
 80     /**
 81      * Return the set flags
 82      * @returns {String[]}
 83      */
 84     this.getSetFlags = function () {
 85         return _setFlags;
 86     };
 87 
 88 
 89     /**
 90      * Returns the set regular expression
 91      * @returns {String}
 92      */
 93     this.getRegularExpression = function () {
 94         return _expression;
 95     };
 96 
 97 
 98     /**
 99      * Parse a text and return result to callback
100      * @param {String} text text to parse
101      * @param {Function} callback Callback for result
102      * @returns this
103      */
104     this.parse  = function (text, callback) {
105 
106         // check in cache for value
107         var cache = RegularExpressionCache.getValue(
108             text,
109             _expression,
110             _setFlags,
111             _name
112         );
113 
114         if (typeof cache != 'boolean') {
115 
116             // found in cache
117             callback(cache);
118             return this;
119         }
120 
121 
122         // otherwise call parser
123         new _parser(this, text, _expression, _setFlags, function(data) {
124 
125             // error handling
126             if (!data || typeof data.error == "undefined") {
127                 return;
128             }
129 
130             // add value to cache
131             RegularExpressionCache.addValue(data);
132 
133             // call original callback
134             callback(data);
135         });
136 
137         return this;
138     };
139 }
