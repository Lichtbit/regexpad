1 /**
  2  * Generates one help step
  3  * Highlight an element
  4  *
  5  * @author Georg Limbach <georf@dev.mgvmedia.com>
  6  *
  7  * @constructor
  8  * @param {String} element jquery selector string
  9  */
 10 function HighlightStep(element) {
 11     "use strict";
 12 
 13     /**
 14      * run method
 15      * @type Function
 16      */
 17     this.run = function() {
 18         $(element).css({position: 'relative', 'z-index': '999'});
 19     };
 20 
 21 
 22     /**
 23      * time for this step
 24      * @type int
 25      */
 26     this.time = 100;
 27 }
