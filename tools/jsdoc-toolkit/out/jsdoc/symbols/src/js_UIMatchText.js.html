1 /**
  2  *
  3  * @author Georg Limbach <georf@dev.mgvmedia.com>
  4  *
  5  * @constructor
  6  * @param jQuery
  7  *            selection - the matchtext block
  8  * @param boolean
  9  *            create "delete" button
 10  */
 11 function UIMatchText($block, deleteButton) {
 12     "use strict";
 13 
 14     // pointer to myself
 15     var self = this;
 16 
 17     // set pointer to special areas
 18     var $textarea = $block.find('.textarea');
 19     var $matchtextDiv = $block.find('.matchtext-div');
 20     var $matchInfo = $block.find('.match-info');
 21     var $nextMatch = $block.find('.next-match');
 22     var $previousMatch = $block.find('.previous-match');
 23     var $matchMoreInfo = $block.find('.match-more-info');
 24     var $moreInformation = $block.find('.more-information');
 25     var $moreInformationContent = $moreInformation.find('.more-information-content');
 26 
 27     // hide boxes
 28     $matchMoreInfo.hide();
 29     $moreInformation.hide();
 30 
 31     // generate model
 32     var matchText = RegHex.addMatchText(this, $textarea.val());
 33 
 34     // Tell the registered match text about changes
 35     $textarea.bind('change keyup paste cut', function() {
 36         matchText.notify();
 37     });
 38 
 39     // get scroll events
 40     $textarea.scroll(function() {
 41         self.updateScrollPosition();
 42     });
 43 
 44     // last response
 45     var _response = null;
 46 
 47     // position of selected match
 48     var _selectedMatch = 0;
 49 
 50     // is the "more information" dialog open?
 51     var _moreInformationOpen = false;
 52 
 53 
 54     /**
 55      * Escape special chars
 56      *
 57      * @param String
 58      *            text to escape
 59      * @return String escaped text
 60      */
 61     var _escapeHtml = function(text) {
 62         text = replaceAll(text, '<', '<');
 63         text = replaceAll(text, '>', '>');
 64         text = replaceAll(text, '\n', '<br>');
 65         return text;
 66     };
 67 
 68 
 69     /**
 70      * Replaces each substring of the value that matches the given string with
 71      * the given replacement.
 72      *
 73      * @param String
 74      *            value
 75      * @param String
 76      *            search for
 77      * @param String
 78      *            replace with
 79      * @return String new text
 80      */
 81     var replaceAll = function(value, find, replace) {
 82         while (value.indexOf(find) != -1) {
 83             value = value.replace(find, replace);
 84         }
 85         return value;
 86     };
 87 
 88 
 89     /**
 90      * update the match text after an new response
 91      *
 92      * @param {object} response
 93      * @return this
 94      */
 95     this.notify = function(response) {
 96 
 97         // its the same text?
 98         if ($textarea.val() == response.matchText) {
 99 
100             // is it a new response?
101             if (_response != response) {
102 
103                 // set new response
104                 _response = response;
105 
106                 // set match to 0
107                 _selectedMatch = 0;
108 
109                 // update all information
110                 this.update();
111             }
112         }
113 
114         return this;
115     };
116 
117 
118     /**
119      * Highlight the matches in the textarea
120      *
121      * @return this
122      */
123     this.highlight = function() {
124 
125         // on error highlight nothing
126         if (!_response || _response.error) {
127             $matchtextDiv.html('');
128 
129         // otherwise highlight selected part
130         } else {
131 
132             var text = _response.matchText;
133             var output = "";
134 
135             // no matches
136             if (_response.matchings.length == 0) {
137 
138                 // all unmatching
139                 output = '<span class="unmatched">' + _escapeHtml(text) + '</span>';
140 
141             } else {
142 
143                 var length = _response.matchings[_selectedMatch].text.length;
144                 var pos = _response.matchings[_selectedMatch].index;
145 
146                 if (length <= 0) {
147                     output = '<span class="unmatched">' + _escapeHtml(text) + '</span>';
148                 } else {
149 
150                     if (pos != 0) {
151                         output +=
152                             '<span class="unmatched">' +
153                                 _escapeHtml(text.substring(0, pos)) +
154                             '</span>';
155                     }
156 
157                     output +=
158                         '<span class="matched">' +
159                             _escapeHtml(text.substring(pos, pos + length)) +
160                         '</span>';
161 
162                     if (text.length != pos + length) {
163                         output +=
164                             '<span class="unmatched">' +
165                                 _escapeHtml(text.substring(pos + length)) +
166                             '</span>';
167                     }
168                 }
169             }
170 
171             // add a no-break space if last char is a new line
172             if (text.length != 0 && text.substring(text.length - 1) == "\n") {
173                 output += '<span> </span>';
174             }
175 
176             // set spans to background area
177             $matchtextDiv.html(output);
178         }
179 
180         return this;
181     };
182 
183 
184     /**
185      * returns the current value of the textarea
186      * @return {String}
187      */
188     this.getText = function() {
189         return $textarea.val();
190     };
191 
192 
193     /**
194      * Sets the current text
195      * @param {String}
196      * @return this
197      */
198     this.setText = function (text) {
199         $textarea.val(text).change();
200         return this;
201     };
202 
203 
204     /**
205      * Updates the infobox
206      *
207      * Sets button dis/enabled and update text
208      */
209     this.updateInfobox = function() {
210 
211         if (!_moreInformationOpen) {
212             $moreInformation.slideUp();
213         }
214 
215         // if its no valid response or if no matchings
216         if (!_response || _response.error || _response.matchings.length == 0) {
217             $matchInfo.html('no matches');
218             $nextMatch.attr('disabled', 'disabled');
219             $previousMatch.attr('disabled', 'disabled');
220 
221             $matchMoreInfo.slideUp();
222             $moreInformation.slideUp();
223 
224         // valid matchings
225         } else {
226             $matchInfo.html('match ' + (_selectedMatch + 1) + ' of ' + _response.matchings.length);
227 
228             var results = _response.matchings[_selectedMatch].subexpressions;
229 
230             // no subexpressions
231             if (!results || results.length <= 0) {
232                 $moreInformationContent.html('no subexpressions given.');
233 
234 
235             // show subexpressions
236             } else {
237 
238                 var output = 'subexpressions <small>(expression parts in brackets)</small>:<ul>';
239                 for ( var i = 0; i < results.length; i++) {
240                     output += '<li>' + $('<li/>').text('$' + i + ' = ' + results[i]).html() + '</li>';
241                 }
242                 output += '</ul>';
243 
244                 $moreInformationContent.html(output);
245             }
246 
247             // disable/enable match buttons
248             if (_selectedMatch == 0 || _response.error) {
249                 $previousMatch.attr('disabled', 'disabled');
250             } else {
251                 $previousMatch.removeAttr('disabled');
252             }
253 
254             if (_selectedMatch >= _response.matchings.length - 1
255             || _response.error) {
256                 $nextMatch.attr('disabled', 'disabled');
257             } else {
258                 $nextMatch.removeAttr('disabled');
259             }
260 
261             // show more information
262             if (_moreInformationOpen) {
263                 $moreInformation.slideDown();
264             } else {
265                 $matchMoreInfo.slideDown();
266             }
267         }
268     };
269 
270 
271     /**
272      * Updates all information in and around the textbox
273      *
274      * @return this
275      */
276     this.update = function() {
277         // update highlighting
278         this.highlight();
279 
280         // update infobox
281         this.updateInfobox();
282 
283         // update scroll position
284         this.updateScrollPosition();
285 
286         return this;
287     };
288 
289 
290     /**
291      * Scroll the background div to the correct position
292      *
293      * @return this
294      */
295     this.updateScrollPosition = function() {
296         $matchtextDiv.scrollTop($textarea.scrollTop());
297         $matchtextDiv.scrollLeft($textarea.scrollLeft());
298 
299         return this;
300     };
301 
302 
303     /**
304      * Select next match if possible
305      *
306      * @return this
307      */
308     this.nextMatch = function() {
309         if (_response
310                 && _selectedMatch < _response.matchings.length - 1) {
311             _selectedMatch++;
312         }
313         this.update();
314 
315         return this;
316     };
317 
318 
319     /**
320      * Select previous match if possible
321      *
322      * @return this
323      */
324     this.previousMatch = function() {
325         if (_selectedMatch > 0) {
326             _selectedMatch--;
327         }
328         this.update();
329 
330         return this;
331     };
332 
333 
334     /**
335      * Open and close the information box
336      *
337      * @param boolean
338      *            open == true
339      * @return this
340      */
341     this.moreInformation = function(open) {
342         if (open == undefined) {
343           open = !_moreInformationOpen;
344         }
345         if (_moreInformationOpen != open) {
346             _moreInformationOpen = open;
347             this.update();
348         }
349         return this;
350     };
351 
352 
353     // set button handler
354     $nextMatch.click(function() {
355         self.nextMatch();
356     });
357     $previousMatch.click(function() {
358         self.previousMatch();
359     });
360 
361     $matchMoreInfo.find('a').click(function() {
362         self.moreInformation();
363         return false;
364     });
365     $moreInformation.find('.less-information').click(function() {
366         self.moreInformation(false);
367         return false;
368     });
369 
370     // update all informations after first creation
371     this.update();
372 
373     if (deleteButton) {
374 
375         // generate delete button
376         var $btn = $('<button type="button" class="remove-matchtext"/>');
377         $btn.html('–');
378         $btn.attr('title', 'remove field');
379         $btn.click(function() {
380             $block.slideUp(function() {
381                     // remove from RegHex
382                     RegHex.removeMatchText(matchText);
383                     $(this).remove();
384                 });
385         });
386 
387         // add to block
388         $block.find('.navigate-match-section').prepend(' ').prepend($btn);
389         $block.hide();
390         $('#matching-blocks').append($block);
391         $block.show();
392     }
393 }
