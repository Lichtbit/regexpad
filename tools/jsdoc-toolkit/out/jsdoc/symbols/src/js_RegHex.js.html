1 /**
  2  * Dispatches tasks and provides Reghex's component objects
  3  *
  4  * @author Georg Limbach <georf@dev.mgvmedia.com>
  5  * @author Sebastian Gaul <sebastian@dev.mgvmedia.com>
  6  *
  7  * @class
  8  */
  9 var RegHex = new function() {
 10     "use strict";
 11 
 12     // pointer to myself
 13     var self = this;
 14 
 15 
 16     /**
 17      * Pointer to select regular expression
 18      * @type RegularExpression
 19      */
 20     var _regularExpression;
 21 
 22 
 23     /**
 24      * Array of MatchText
 25      * @type MatchText[]
 26      */
 27     var _matchTexts = [];
 28 
 29 
 30     /**
 31      * Array of change handler
 32      * @type function[]
 33      */
 34     var _changeHandlers = [];
 35 
 36 
 37     /**
 38      * Object for message service
 39      *
 40      * Initialitation with senseless function
 41      * @type {UIMessageService}
 42      */
 43     var _messageService = { notify: function() {} };
 44 
 45 
 46     /**
 47      * Object for presenting code
 48      *
 49      * @type {UICodeSnip}
 50      */
 51     var _codeSnip = { update: function(){}};
 52 
 53 
 54     /**
 55      * Update all match texts
 56      */
 57     var _updateMatchTexts = function() {
 58         var errorOccured = false;
 59         for ( var i = 0; i < _matchTexts.length; i++) {
 60             _matchTexts[i].notify(_matchTexts[i].observer,
 61                 function(error) {
 62                     _messageService.notify(error);
 63                     errorOccured = true;
 64                 });
 65         }
 66 
 67 
 68         // Notify message service that parsing was successful, so it
 69         // can hide previous error messages
 70         if (!errorOccured) {
 71             _messageService.notify();
 72         }
 73     };
 74 
 75 
 76     /**
 77      * Update the code snip and change handlers about response
 78      * @param {Object}
 79      */
 80     var _response = function(response) {
 81         var i, l;
 82 
 83         if (typeof response.programming != 'undefined') {
 84             _codeSnip.update(response.programming);
 85         }
 86 
 87         for (i = 0, l = _changeHandlers.length; i < l; i++) {
 88             _changeHandlers[i].call();
 89         }
 90     };
 91 
 92 
 93     /**
 94      * Creates a new match text and registers a specified observer
 95      *
 96      * @param {UIMatchText} observer
 97      * @param {String} initialText
 98      * @returns {MatchText} The created MatchText object
 99      */
100     this.addMatchText = function(observer, initialText) {
101 
102         // create object
103         var matchText = new MatchText(observer);
104 
105         // set text
106         matchText.setText(initialText);
107 
108         // set default response callback
109         matchText.setResponseCallback(_response);
110 
111         // add to list
112         _matchTexts.push(matchText);
113 
114         return matchText;
115     };
116 
117 
118     /**
119      * Returns the match texts as an array
120      * @returns {MatchText[]}
121      */
122     this.getMatchTexts = function () {
123         return _matchTexts;
124     };
125 
126 
127     /**
128      * Remove a match text
129      *
130      * @param {MatchText} matchText to remove
131      * @return this
132      */
133     this.removeMatchText = function(matchText) {
134         for ( var i = 0; i < _matchTexts.length; i++) {
135             if (_matchTexts[i] == matchText) {
136                 var current = _matchTexts[i];
137 
138                 // remove from array
139                 _matchTexts.splice(i, 1);
140                 break;
141             }
142         }
143         return this;
144     };
145 
146 
147     /**
148      * Registers a message service which accepts and handles messages
149      *
150      * @param {UIMessageService} messageService
151      */
152     this.registerMessageService = function(messageService) {
153         _messageService = messageService;
154     };
155 
156 
157     /**
158      * Registers a code snip presenter
159      *
160      * @param {UICodeSnip} codeSnip
161      */
162     this.registerCodeSnip = function(codeSnip) {
163         _codeSnip = codeSnip;
164     };
165 
166 
167     /**
168      * Updates the reg. exp., e.g. after the user changed the related field
169      *
170      * @param {String} expression e.g. "ab*c"
171      * @param {String[]} flags e.g. ["i", "x", "s"]
172      */
173     this.updateRegularExpression = function(expression, flags) {
174         // set new expression
175         _regularExpression.setRegularExpression(expression);
176 
177         // set new flags
178         _regularExpression.setFlags(flags);
179 
180         // update match texts
181         _updateMatchTexts();
182     };
183 
184 
185     /**
186      * Returns the RegularExpression object
187      *
188      * @return RegularExpression
189      */
190     this.getRegularExpression = function() {
191         return _regularExpression;
192     };
193 
194 
195     /**
196      * Change the parser type e.g. the user select a other parser
197      *
198      * The method search the object for the new parser and set is as
199      * current regular expression object.
200      *
201      * @param {String} parserType
202      */
203     this.changeParserType = function(parserType) {
204 
205         // search config in array
206         for (var i = 0; i < config.parsers.length; i++) {
207             if (config.parsers[i].getName() == parserType) {
208                 _regularExpression = config.parsers[i];
209             }
210         }
211 
212         // update match texts
213         _updateMatchTexts();
214     };
215 
216 
217     /**
218      * registers a change handler
219      *
220      * @param {function} callback
221      */
222     this.registerChangeHandler = function(callback) {
223         _changeHandlers.push(callback);
224     };
225 };
